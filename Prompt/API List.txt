        >
          <X className="w-6 h-6 text-white" />
        </button>
        {children}
      </div>
    </div>
  );
};

const CopyAddress = ({ label, address }) => {
  const [copied, setCopied] = useState(false);
  
  const handleCopy = () => {
    navigator.clipboard.writeText(address); // May not work in iframe, using simple feedback
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  return (
    <div className="w-full">
      <div className="flex justify-between text-xs text-slate-400 mb-1">
        <span>{label}</span>
        {copied && <span className="text-green-400 flex items-center gap-1"><CheckCircle className="w-3 h-3"/> Copied</span>}
      </div>
      <button 
        onClick={handleCopy}
        className="w-full flex items-center justify-between bg-slate-900 border border-slate-700 hover:border-cyan-500/50 rounded p-3 text-sm text-slate-300 transition-all group"
      >
        <span className="font-mono truncate mr-2">{address.substring(0, 20)}...{address.substring(address.length - 4)}</span>
        <Copy className="w-4 h-4 text-slate-500 group-hover:text-cyan-400" />
      </button>
    </div>
  );
};
